// Vas kod ide ovdje

int main() {
        Datum d(15, 4, 2021);
        Vrijeme v(23, 59);

        d.Ispisi();
        std::cout << std::endl;
        v.Ispisi();
        std::cout << std::endl;

        try
        {
            d.Postavi(-1, 2, 2023);
        }
        catch(const std::domain_error& e)
        {
            std::cout << e.what() << '\n';
        }

        try
        {
            v.Postavi(25, 10);
        }
        catch (const std::domain_error &e)
        {
            std::cout << e.what() << '\n';
        }

        Pregled p("Neko Nekić", d, v);
        p.Ispisi();

        Pregledi pregledi(0);
        try {
            pregledi.RegistrirajPregled("Neko Nekić", 1, 1, 2022, 8, 0);
        } catch (const std::range_error &e) {
            std::cout << e.what() << '\n';
        }
        try
        {
            p = pregledi.DajNajranijiPregled();
        }
        catch (const std::domain_error &e)
        {
            std::cout << e.what() << '\n';
        }
        try
        {
            pregledi.ObrisiNajranijiPregled();
        }
        catch (const std::range_error &e)
        {
            std::cout << e.what() << '\n';
        }

    return 0;
}
